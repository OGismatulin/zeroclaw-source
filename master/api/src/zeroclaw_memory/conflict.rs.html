1//! Conflict resolution for memory entries.
2//!
3//! On the consolidation path (#18), `judge_conflicts` asks an LLM which existing
4//! Core facts a new fact makes obsolete (direct contradiction / same-attribute
5//! replacement), then those are marked superseded. This replaced the older
6//! similarity-threshold check (`check_and_resolve_conflicts`, retained below for
7//! any other/no callers) — a measurement showed cosine similarity cannot cleanly
8//! separate contradiction from mere relatedness.
9
10use super::traits::{Memory, MemoryCategory, MemoryEntry};
11use zeroclaw_api::model_provider::ModelProvider;

13/// System prompt for the Core-dedup LLM judge (#18).
14///
15/// Conservative on purpose: a measurement showed cosine similarity cannot
16/// cleanly separate "contradiction" from "mere relatedness" (cos between
17/// "language is Python" and "language is Rust" was only 0.78). The judge is
18/// asked to mark ONLY facts whose attribute value the new fact replaces.
19///
20/// The marker word "устаревшими" is what the test stub matches on to detect
21/// the judge prompt — keep it present.
22const JUDGE_SYSTEM_PROMPT: &str = r#"Ты — строгий арбитр памяти. Тебе дают НОВЫЙ факт о пользователе и пронумерованный список СУЩЕСТВУЮЩИХ фактов.
23
24Определи, какие из существующих фактов новый факт делает устаревшими — то есть прямо противоречит им или заменяет значение того же самого атрибута (например, сменился основной язык программирования, город, должность, имя, предпочтение).
25
26ПРАВИЛА:
27- Помечай факт только если новый факт ЗАМЕНЯЕТ значение того же атрибута или прямо ему противоречит.
28- НЕ помечай факты, которые просто связаны по теме, дополняют новый факт или относятся к другому атрибуту.
29- Если ни один факт не устарел — верни пустой массив.
30
31Ответь СТРОГО JSON-массивом номеров устаревших фактов, например: [2]
32Если устаревших нет: []
33Не добавляй никакого текста вне JSON-массива."#;

35/// Ask an LLM which existing Core facts the new fact makes outdated (#18).
36///
37/// Returns the `id`s of `candidates` the judge marks as superseded. This is a
38/// safe, best-effort operation: any provider error or unparseable reply yields
39/// an empty Vec (supersede nothing) — it never propagates an error, never
40/// panics, and never supersedes on doubt.
41pub async fn judge_conflicts(
42    provider: &dyn ModelProvider,
43    model: &str,
44    temperature: Option<f64>,
45    new_content: &str,
46    candidates: &[MemoryEntry],
47) -> anyhow::Result<Vec<String>> {
48    if candidates.is_empty() {
49        return Ok(Vec::new());
50    }
51
52    // Build a 1-based numbered list of existing facts.
53    let mut list = String::new();
54    for (i, c) in candidates.iter().enumerate() {
55        list.push_str(&format!("{}. {}\n", i + 1, c.content));
56    }
57    let msg = format!(
58        "НОВЫЙ факт:\n{new_content}\n\nСУЩЕСТВУЮЩИЕ факты:\n{list}\nКакие существующие факты устарели?"
59    );
60
61    let raw = match provider
62        .chat_with_system(Some(JUDGE_SYSTEM_PROMPT), &msg, model, temperature)
63        .await
64    {
65        Ok(r) => r,
66        Err(e) => {
67            ::zeroclaw_log::record!(
68                DEBUG,
69                ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Note)
70                    .with_attrs(::serde_json::json!({"error": format!("{}", e)})),
71                "judge_conflicts provider call failed; superseding nothing"
72            );
73            return Ok(Vec::new());
74        }
75    };
76
77    // Tolerant parse: slice from the first `[` to the LAST `]` (widest span,
78    // tolerates surrounding prose) and parse as Vec<i64>. Not a balanced-bracket
79    // parser: a nested array in that span fails to parse and yields []. Any
80    // failure is safe here — it means "supersede nothing".
81    let indices: Vec<i64> = match (raw.find('['), raw.rfind(']')) {
82        (Some(start), Some(end)) if end > start => {
83            serde_json::from_str::<Vec<i64>>(&raw[start..=end]).unwrap_or_default()
84        }
85        _ => Vec::new(),
86    };
87
88    // Map valid 1-based indices in range to candidate ids.
89    let mut superseded = Vec::new();
90    for n in indices {
91        if n >= 1 && (n as usize) <= candidates.len() {
92            superseded.push(candidates[(n as usize) - 1].id.clone());
93        }
94    }
95
96    Ok(superseded)
97}

99/// Check for conflicting memories and mark old ones as superseded.
100///
101/// Returns the list of entry IDs that were superseded.
102// NOTE: superseded by judge_conflicts on the consolidation path (#18); retained for any other/no callers.
103pub async fn check_and_resolve_conflicts(
104    memory: &dyn Memory,
105    key: &str,
106    content: &str,
107    category: &MemoryCategory,
108    threshold: f64,
109) -> anyhow::Result<Vec<String>> {
110    // Only check conflicts for Core memories
111    if !matches!(category, MemoryCategory::Core) {
112        return Ok(Vec::new());
113    }
114
115    // Search for similar existing entries
116    let candidates = memory.recall(content, 10, None, None, None).await?;
117
118    let mut superseded = Vec::new();
119    for candidate in &candidates {
120        if candidate.key == key {
121            continue; // Same key = update, not conflict
122        }
123        if !matches!(candidate.category, MemoryCategory::Core) {
124            continue;
125        }
126        if let Some(score) = candidate.score
127            && score > threshold
128            && candidate.content != content
129        {
130            superseded.push(candidate.id.clone());
131        }
132    }
133
134    Ok(superseded)
135}

137/// Mark entries as superseded in SQLite by setting their `superseded_by` column.
138pub fn mark_superseded(
139    conn: &rusqlite::Connection,
140    superseded_ids: &[String],
141    new_id: &str,
142) -> anyhow::Result<()> {
143    if superseded_ids.is_empty() {
144        return Ok(());
145    }
146
147    for id in superseded_ids {
148        conn.execute(
149            "UPDATE memories SET superseded_by = ?1 WHERE id = ?2",
150            rusqlite::params![new_id, id],
151        )?;
152    }
153
154    Ok(())
155}

157/// Simple text-based conflict detection without embeddings.
158///
159/// Uses token overlap (Jaccard similarity) as a fast approximation
160/// when vector embeddings are unavailable.
161pub fn jaccard_similarity(a: &str, b: &str) -> f64 {
162    let words_a: std::collections::HashSet<&str> = a.split_whitespace().collect();
163    let words_b: std::collections::HashSet<&str> = b.split_whitespace().collect();
164
165    if words_a.is_empty() && words_b.is_empty() {
166        return 1.0;
167    }
168    if words_a.is_empty() || words_b.is_empty() {
169        return 0.0;
170    }
171
172    let intersection = words_a.intersection(&words_b).count();
173    let union = words_a.union(&words_b).count();
174
175    if union == 0 {
176        0.0
177    } else {
178        intersection as f64 / union as f64
179    }
180}

182/// Find potentially conflicting entries using text similarity when embeddings
183/// are not available. Returns entries above the threshold.
184pub fn find_text_conflicts(
185    entries: &[MemoryEntry],
186    new_content: &str,
187    threshold: f64,
188) -> Vec<String> {
189    entries
190        .iter()
191        .filter(|e| {
192            matches!(e.category, MemoryCategory::Core)
193                && e.superseded_by.is_none()
194                && jaccard_similarity(&e.content, new_content) > threshold
195                && e.content != new_content
196        })
197        .map(|e| e.id.clone())
198        .collect()
199}
