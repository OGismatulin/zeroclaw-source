1//! LLM-driven memory consolidation.
2//!
3//! After each conversation turn, extracts structured information:
4//! - `history_entry`: A timestamped summary for the daily conversation log.
5//! - `memory_update`: New facts, preferences, or decisions worth remembering
6//!   long-term (or `null` if nothing new was learned).
7//!
8//! This two-phase approach replaces the naive raw-message auto-save with
9//! semantic extraction, similar to Nanobot's `save_memory` tool call pattern.
10
11use crate::conflict;
12use crate::importance;
13use crate::traits::{Memory, MemoryCategory};
14use zeroclaw_api::model_provider::ModelProvider;
15
16/// Output of consolidation extraction.
17#[derive(Debug, serde::Deserialize)]
18pub struct ConsolidationResult {
19    /// Brief timestamped summary for the conversation history log.
20    pub history_entry: String,
21    /// New facts/preferences/decisions to store long-term, or None.
22    pub memory_update: Option<String>,
23    /// Atomic facts extracted from the turn (when consolidation_extract_facts is enabled).
24    #[serde(default)]
25    pub facts: Vec<String>,
26    /// Observed trend or pattern (when consolidation_extract_facts is enabled).
27    #[serde(default)]
28    pub trend: Option<String>,
29}

31// Fork patch: prompt rewritten so extracted facts come back in the
32// conversation's language (this deployment is Russian-centric). EN facts in a
33// RU context surface in `[Memory context]` of RU turns and degrade recall
34// (EN-fact ↔ RU-query embedding similarity is lower).
35const CONSOLIDATION_SYSTEM_PROMPT: &str = r#"Ты — движок консолидации памяти. По одному ходу диалога извлеки:
361. "history_entry": краткое summary того, что произошло в этом ходе (1–2 предложения). Укажи ключевую тему или действие.
372. "memory_update": любые НОВЫЕ факты, предпочтения, решения или обязательства, которые стоит запомнить надолго. Верни null, если ничего нового не узнал.
38
39Пиши history_entry и memory_update на ТОМ ЖЕ ЯЗЫКЕ, на котором идёт диалог (обычно русский).
40Ответь ТОЛЬКО валидным JSON: {"history_entry": "...", "memory_update": "..." или null}
41Не добавляй никакого текста вне JSON-объекта."#;

43/// Run two-phase LLM-driven consolidation on a conversation turn.
44///
45/// Phase 1: Write a history entry to the Daily memory category.
46/// Phase 2: Write a memory update to the Core category (if the LLM identified new facts).
47///
48/// This function is designed to be called fire-and-forget via `zeroclaw_spawn::spawn!`.
49/// Strip channel media markers (e.g. `[IMAGE:/local/path]`, `[DOCUMENT:...]`)
50/// that contain local filesystem paths.  These must never be forwarded to
51/// upstream model_provider APIs — they would leak local paths and cause API errors.
52fn strip_media_markers(text: &str) -> String {
53    // Matches [IMAGE:...], [DOCUMENT:...], [FILE:...], [VIDEO:...], [VOICE:...], [AUDIO:...]
54    static RE: std::sync::LazyLock<regex::Regex> = std::sync::LazyLock::new(|| {
55        regex::Regex::new(r"\[(?:IMAGE|DOCUMENT|FILE|VIDEO|VOICE|AUDIO):[^\]]*\]").unwrap()
56    });
57    RE.replace_all(text, "[media attachment]").into_owned()
58}

60pub async fn consolidate_turn(
61    model_provider: &dyn ModelProvider,
62    model: &str,
63    temperature: Option<f64>,
64    memory: &dyn Memory,
65    user_message: &str,
66    assistant_response: &str,
67) -> anyhow::Result<()> {
68    let turn_text = format!(
69        "User: {}\nAssistant: {}",
70        strip_media_markers(user_message),
71        strip_media_markers(assistant_response),
72    );
73
74    // Truncate very long turns to avoid wasting tokens on consolidation.
75    // Use char-boundary-safe slicing to prevent panic on multi-byte UTF-8 (e.g. CJK text).
76    let truncated = if turn_text.len() > 4000 {
77        let end = turn_text
78            .char_indices()
79            .map(|(i, _)| i)
80            .take_while(|&i| i <= 4000)
81            .last()
82            .unwrap_or(0);
83        format!("{}…", &turn_text[..end])
84    } else {
85        turn_text.clone()
86    };
87
88    let raw = model_provider
89        .chat_with_system(
90            Some(CONSOLIDATION_SYSTEM_PROMPT),
91            &truncated,
92            model,
93            temperature,
94        )
95        .await?;
96
97    let result: ConsolidationResult = parse_consolidation_response(&raw, &turn_text);
98
99    // Phase 1: Write history entry to Daily category.
100    let date = chrono::Local::now().format("%Y-%m-%d").to_string();
101    let history_key = format!("daily_{date}_{}", uuid::Uuid::new_v4());
102    memory
103        .store(
104            &history_key,
105            &result.history_entry,
106            MemoryCategory::Daily,
107            None,
108        )
109        .await?;
110
111    // Phase 2: Write memory update to Core category (if present).
112    if let Some(ref update) = result.memory_update
113        && !update.trim().is_empty()
114    {
115        let mem_key = format!("core_{}", uuid::Uuid::new_v4());
116
117        // Compute importance score heuristically.
118        let imp = importance::compute_importance(update, &MemoryCategory::Core);
119
120        // Detect conflicting Core memories BEFORE storing the new entry, so the
121        // new entry itself is never a candidate for being superseded.
122        //
123        // Fork patch #18: a hardcoded similarity threshold cannot cleanly
124        // separate "contradiction" from "mere relatedness" (cos between
125        // "language is Python" and "language is Rust" was only 0.78). Instead,
126        // recall the nearest Core candidates and let an LLM judge decide which
127        // the new fact makes outdated. `judge_conflicts` is best-effort and
128        // returns empty on any provider/parse error (supersede nothing).
129        let candidates: Vec<_> = memory
130            .recall(update, 10, None, None, None)
131            .await
132            .unwrap_or_default()
133            .into_iter()
134            .filter(|c| {
135                matches!(c.category, MemoryCategory::Core)
136                    && c.key != mem_key
137                    && c.content != *update
138            })
139            .collect();
140        let superseded_ids =
141            conflict::judge_conflicts(model_provider, model, temperature, update, &candidates)
142                .await
143                .unwrap_or_default();
144
145        // Store with importance metadata.
146        memory
147            .store_with_metadata(
148                &mem_key,
149                update,
150                MemoryCategory::Core,
151                None,
152                None,
153                Some(imp),
154            )
155            .await?;
156
157        // Retire the older conflicting facts: recall filters
158        // `superseded_by IS NULL`, so this is what actually de-duplicates Core.
159        // The marker value is the new entry's key — only NULL-ness gates recall
160        // (no FK/JOIN reads the value), and the new row's internal id is not
161        // available here without an extra round-trip.
162        if !superseded_ids.is_empty() {
163            // Observability (#18): record what the judge retired and why.
164            let update_preview: String = update.chars().take(120).collect();
165            let superseded_contents: Vec<String> = superseded_ids
166                .iter()
167                .filter_map(|id| {
168                    candidates
169                        .iter()
170                        .find(|c| &c.id == id)
171                        .map(|c| c.content.chars().take(120).collect::<String>())
172                })
173                .collect();
174            ::zeroclaw_log::record!(
175                DEBUG,
176                ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Note)
177                    .with_attrs(::serde_json::json!({
178                        "new_key": mem_key,
179                        "new_content": update_preview,
180                        "superseded_ids": superseded_ids,
181                        "superseded_contents": superseded_contents,
182                    })),
183                "judge superseded Core facts"
184            );
185
186            let refs: Vec<&str> = superseded_ids.iter().map(|s| s.as_str()).collect();
187            if let Err(e) = memory.mark_superseded(&refs, &mem_key).await {
188                ::zeroclaw_log::record!(
189                    DEBUG,
190                    ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Note)
191                        .with_attrs(::serde_json::json!({"error": format!("{}", e)})),
192                    "mark_superseded skipped"
193                );
194            }
195        }
196    }
197
198    Ok(())
199}

201/// Parse the LLM's consolidation response, with fallback for malformed JSON.
202fn parse_consolidation_response(raw: &str, fallback_text: &str) -> ConsolidationResult {
203    // Try to extract JSON from the response (LLM may wrap in markdown code blocks).
204    let cleaned = raw
205        .trim()
206        .trim_start_matches("```json")
207        .trim_start_matches("```")
208        .trim_end_matches("```")
209        .trim();
210
211    serde_json::from_str(cleaned).unwrap_or_else(|_| {
212        // Fallback: use truncated turn text as history entry.
213        // Use char-boundary-safe slicing to prevent panic on multi-byte UTF-8.
214        let summary = if fallback_text.len() > 200 {
215            let end = fallback_text
216                .char_indices()
217                .map(|(i, _)| i)
218                .take_while(|&i| i <= 200)
219                .last()
220                .unwrap_or(0);
221            format!("{}…", &fallback_text[..end])
222        } else {
223            fallback_text.to_string()
224        };
225        ConsolidationResult {
226            history_entry: summary,
227            memory_update: None,
228            facts: Vec::new(),
229            trend: None,
230        }
231    })
232}

273    #[test]
274    fn fallback_truncates_long_text() {
275        let long_text = "x".repeat(500);
276        let result = parse_consolidation_response("invalid", &long_text);
277        // 200 bytes + "…" (3 bytes in UTF-8) = 203
278        assert!(result.history_entry.len() <= 203);
279    }
280
281    #[test]
282    fn fallback_truncates_cjk_text_without_panic() {
283        // Each CJK character is 3 bytes in UTF-8; byte index 200 may land
284        // inside a character. This must not panic.
285        let cjk_text = "二手书项目".repeat(50); // 250 chars = 750 bytes
286        let result = parse_consolidation_response("invalid", &cjk_text);
287        assert!(
288            result
289                .history_entry
290                .is_char_boundary(result.history_entry.len())
291        );
292        assert!(result.history_entry.ends_with('…'));
293    }

304    /// Returns a canned reply chosen by inspecting the system prompt: the
305    /// consolidation extraction prompt vs the Core-dedup judge prompt. The
306    /// judge prompt is detected by its unique marker word ("устаревшими").
307    /// `chat` keeps its trait default (unused here).
308    struct StubProvider {
309        extraction_reply: String,
310        judge_reply: String,
311    }

313    #[async_trait]
314    impl ModelProvider for StubProvider {
315        async fn chat_with_system(
316            &self,
317            system_prompt: Option<&str>,
318            _message: &str,
319            _model: &str,
320            _temperature: Option<f64>,
321        ) -> anyhow::Result<String> {
322            let is_judge = system_prompt
323                .map(|p| p.contains("устаревшими"))
324                .unwrap_or(false);
325            if is_judge {
326                // Sentinel: pick the 1-based index of the candidate line that
327                // mentions "Python", so the test is independent of recall order.
328                if self.judge_reply == "AUTO_PYTHON" {
329                    for line in _message.lines() {
330                        if line.contains("Python")
331                            && let Some((num, _)) = line.split_once('.')
332                            && let Ok(n) = num.trim().parse::<i64>()
333                        {
334                            return Ok(format!("[{n}]"));
335                        }
336                    }
337                    return Ok("[]".into());
338                }
339                Ok(self.judge_reply.clone())
340            } else {
341                Ok(self.extraction_reply.clone())
342            }
343        }
344    }
