8/// Writes the exact LLM input/output to a per-user JSONL trace file.
9///
10/// Enabled via `ZEROCLAW_PROMPT_TRACE`; raw (no scrub). Panic-safe: every
11/// IO/serialize error is swallowed so a trace failure never aborts a turn.
12pub struct PromptTraceHook {
13    /// `<workspace_dir>/logs/prompt-trace.jsonl`
14    path: PathBuf,
15    max_bytes: u64,
16}
17
18impl PromptTraceHook {
19    pub fn new(workspace_dir: PathBuf, max_bytes: u64) -> Self {
20        let path = workspace_dir.join("logs").join("prompt-trace.jsonl");
21        Self { path, max_bytes }
22    }
23
24    fn now_rfc3339() -> String {
25        chrono::Utc::now().to_rfc3339()
26    }
27
28    /// Rotate keeping 3 backups (.1 -> .2 -> .3) when file >= max_bytes.
29    /// Best-effort: ignores all errors.
30    fn rotate_if_needed(&self) {
31        let too_big = std::fs::metadata(&self.path)
32            .map(|m| m.len() >= self.max_bytes)
33            .unwrap_or(false);
34        if !too_big {
35            return;
36        }
37        let p = |n: u32| self.path.with_extension(format!("jsonl.{n}"));
38        let _ = std::fs::rename(p(2), p(3));
39        let _ = std::fs::rename(p(1), p(2));
40        let _ = std::fs::rename(&self.path, p(1));
41    }
42
43    fn append_line(&self, value: &serde_json::Value) {
44        use std::io::Write;
45        if let Some(parent) = self.path.parent() {
46            let _ = std::fs::create_dir_all(parent);
47        }
48        self.rotate_if_needed();
49        let line = match serde_json::to_string(value) {
50            Ok(s) => s,
51            Err(_) => return,
52        };
53        if let Ok(mut f) = std::fs::OpenOptions::new()
54            .create(true)
55            .append(true)
56            .open(&self.path)
57        {
58            let _ = writeln!(f, "{line}");
59        }
60    }
61}
62
63#[async_trait]
64impl HookHandler for PromptTraceHook {
65    fn name(&self) -> &str {
66        "prompt_trace"
67    }
68
69    async fn on_llm_input(&self, messages: &[ChatMessage], model: &str) {
70        // messages == &prepared_messages.messages, model == active_model (set by caller).
71        let msgs = match serde_json::to_value(messages) {
72            Ok(v) => v,
73            Err(_) => return,
74        };
75        let line = serde_json::json!({
76            "kind": "input",
77            "ts": Self::now_rfc3339(),
78            "model": model,
79            "messages_count": messages.len(),
80            "messages": msgs,
81        });
82        self.append_line(&line);
83    }
84
85    async fn on_llm_output(&self, response: &ChatResponse) {
86        // ChatResponse/TokenUsage have NO Serialize derive — build JSON manually.
87        // `response` is borrowed and the fields are not Copy, so clone what we keep.
88        let line = serde_json::json!({
89            "kind": "output",
90            "ts": Self::now_rfc3339(),
91            "response": {
92                "text": response.text.clone(),
93                "tool_calls": response.tool_calls.clone(),
94                "reasoning_content": response.reasoning_content.clone(),
95                "usage": response.usage.as_ref().map(|u| serde_json::json!({
96                    "input_tokens": u.input_tokens,
97                    "output_tokens": u.output_tokens,
98                    "cached_input_tokens": u.cached_input_tokens,
99                })),
100            },
101        });
102        self.append_line(&line);
103    }
104}

110    #[tokio::test]
111    async fn writes_input_line_as_jsonl() {
112        let dir = tempfile::tempdir().unwrap();
113        let hook = PromptTraceHook::new(dir.path().to_path_buf(), 50 * 1024 * 1024);
114        let msgs = vec![ChatMessage::system("SYS"), ChatMessage::user("hello")];
115        hook.on_llm_input(&msgs, "deepseek-v4-flash").await;
116
117        let path = dir.path().join("logs").join("prompt-trace.jsonl");
118        let body = std::fs::read_to_string(&path).unwrap();
119        let v: serde_json::Value = serde_json::from_str(body.lines().next().unwrap()).unwrap();
120        assert_eq!(v["kind"], "input");
121        assert_eq!(v["model"], "deepseek-v4-flash");
122        assert_eq!(v["messages_count"], 2);
123        assert_eq!(v["messages"][0]["role"], "system");
124        assert_eq!(v["messages"][1]["content"], "hello");
125    }
126
127    #[tokio::test]
128    async fn writes_output_line_without_model_field() {
129        let dir = tempfile::tempdir().unwrap();
130        let hook = PromptTraceHook::new(dir.path().to_path_buf(), 50 * 1024 * 1024);
131        let resp = ChatResponse {
132            text: Some("hi".into()),
133            tool_calls: vec![],
134            usage: None,
135            reasoning_content: None,
136        };
137        hook.on_llm_output(&resp).await;
138        let body = std::fs::read_to_string(dir.path().join("logs/prompt-trace.jsonl")).unwrap();
139        let v: serde_json::Value = serde_json::from_str(body.lines().last().unwrap()).unwrap();
140        assert_eq!(v["kind"], "output");
141        assert!(v.get("model").is_none());
142        assert_eq!(v["response"]["text"], "hi");
143    }

145    #[tokio::test]
146    async fn output_serializes_usage_when_present() {
147        use zeroclaw_api::model_provider::TokenUsage;
148        let dir = tempfile::tempdir().unwrap();
149        let hook = PromptTraceHook::new(dir.path().to_path_buf(), 50 * 1024 * 1024);
150        let resp = ChatResponse {
151            text: None,
152            tool_calls: vec![],
153            usage: Some(TokenUsage {

158            reasoning_content: None,

162        let v: serde_json::Value = serde_json::from_str(body.lines().last().unwrap()).unwrap();
163        assert_eq!(v["response"]["usage"]["input_tokens"], 123);

168    #[tokio::test]
169    async fn rotates_at_threshold_keeping_three_backups() {
170        let dir = tempfile::tempdir().unwrap();
171        // 1 byte threshold => every write after the first triggers a rotation.
172        let hook = PromptTraceHook::new(dir.path().to_path_buf(), 1);
173        let msgs = vec![ChatMessage::user("x")];
174        for _ in 0..3 {
175            hook.on_llm_input(&msgs, "m").await;
176        }
177        assert!(dir.path().join("logs/prompt-trace.jsonl.1").exists());
178    }
