1//! Compatibility shim for the doctor command's log-reading utilities, plus
2//! the fork's legacy positional-arg `record_event` emission surface.
3//!
4//! Upstream retired `record_event` in favor of direct `zeroclaw_log::record!`
5//! invocations with the new ECS-flavored `LogEvent` schema. This fork keeps
6//! the legacy emitter alongside the new shim because external consumers parse
7//! the legacy JSONL shape (`event_type` / `timestamp` / `payload`) from
8//! `runtime-trace.jsonl`:
9//!
10//! - `scripts/gateway_manager.py` ProgressNotifier watchdog reads our
11//!   `context_state` events (`tokens_before`, `context_window`, `percent`,
12//!   `passes`, `session_id`),
13//! - the nightly-retrospective skill's `extract_day.py`,
14//! - the prompt-trace tooling.
15//!
16//! New-format lines written by `zeroclaw_log` and legacy lines written here
17//! coexist in the same file: the `zeroclaw_log` reader skips lines that fail
18//! `LogEvent` parse, rolling rotation on both sides is line-count based, and
19//! `migrate_legacy_jsonl_in_place` only fires when the FIRST line of the file
20//! is legacy-shaped (one-shot conversion of pre-merge history).

46/// Initialize log persistence from the observability config.
47///
48/// Fork: also installs the legacy trace logger so `record_event` /
49/// `record_turn_cancelled` keep appending legacy-shape lines to the same
50/// JSONL file.
51pub fn init_from_config(
52    config: &zeroclaw_config::schema::ObservabilityConfig,
53    workspace_dir: &Path,
54) {
55    zeroclaw_log::init_from_config(&to_log_config(config), workspace_dir);
56
57    let mode = LegacyStorageMode::from_raw(&config.log_persistence);
58    let path = resolve_trace_path(config, workspace_dir);
59    let logger = match mode {
60        LegacyStorageMode::None => None,
61        _ => Some(Arc::new(LegacyTraceLogger::new(
62            mode,
63            config.log_persistence_max_entries,
64            path,
65        ))),
66    };
67    let mut guard = TRACE_LOGGER.write().unwrap_or_else(|e| e.into_inner());
68    *guard = logger;
69}

71/// Resolve the configured log path (used by the doctor command).
72pub fn resolve_trace_path(
73    config: &zeroclaw_config::schema::ObservabilityConfig,
74    workspace_dir: &Path,
75) -> std::path::PathBuf {
76    let policy = zeroclaw_log::ResolvedPolicy::from_config(&to_log_config(config), workspace_dir);
77    policy.path
78}

80/// Load a page of events. Replaces the old `load_events` shape with a
81/// thin wrapper around the new paginated reader. The legacy
82/// `event_filter` (single action match) and `contains` (substring) args
83/// map straight onto the new [`LogFilter`] fields.
84pub fn load_events(
85    path: &Path,
86    limit: usize,
87    event_filter: Option<&str>,
88    contains: Option<&str>,
89) -> anyhow::Result<Vec<LogEvent>> {
90    let filter = LogFilter {
91        action: event_filter.map(str::to_string),
92        q: contains.map(str::to_string),
93        ..LogFilter::default()
94    };
95    let page = zeroclaw_log::load_page(path, &filter, limit)?;
96    Ok(page.events)
97}

108#[derive(Debug, Clone, Copy, PartialEq, Eq)]
109enum LegacyStorageMode {
110    None,
111    Rolling,
112    Full,
113}
114
115impl LegacyStorageMode {
116    fn from_raw(raw: &str) -> Self {
117        match raw.trim().to_ascii_lowercase().as_str() {
118            "rolling" => Self::Rolling,
119            "full" => Self::Full,
120            _ => Self::None,
121        }
122    }
123}

125/// Legacy-shape trace event. Field set and serialization MUST stay
126/// byte-compatible with the pre-v0.8.0 `RuntimeTraceEvent`: external Python
127/// consumers parse these exact keys.
128#[derive(Debug, Clone, Serialize, Deserialize)]
129struct LegacyTraceEvent {
130    pub id: String,
131    pub timestamp: String,
132    pub event_type: String,
133    #[serde(skip_serializing_if = "Option::is_none")]
134    pub channel: Option<String>,
135    #[serde(skip_serializing_if = "Option::is_none")]
136    pub provider: Option<String>,
137    #[serde(skip_serializing_if = "Option::is_none")]
138    pub model: Option<String>,
139    #[serde(skip_serializing_if = "Option::is_none")]
140    pub turn_id: Option<String>,
141    #[serde(skip_serializing_if = "Option::is_none")]
142    pub success: Option<bool>,
143    #[serde(skip_serializing_if = "Option::is_none")]
144    pub message: Option<String>,
145    #[serde(default)]
146    pub payload: Value,
147}
148
149struct LegacyTraceLogger {
150    mode: LegacyStorageMode,
151    max_entries: usize,
152    path: PathBuf,
153    write_lock: std::sync::Mutex<()>,
154}
155
156impl LegacyTraceLogger {
157    fn new(mode: LegacyStorageMode, max_entries: usize, path: PathBuf) -> Self {
158        Self {
159            mode,
160            max_entries: max_entries.max(1),
161            path,
162            write_lock: std::sync::Mutex::new(()),
163        }
164    }
165
166    fn append(&self, event: &LegacyTraceEvent) -> anyhow::Result<()> {
167        if self.mode == LegacyStorageMode::None {
168            return Ok(());
169        }
170
171        let _guard = self.write_lock.lock().unwrap_or_else(|e| e.into_inner());
172
173        if let Some(parent) = self.path.parent() {
174            fs::create_dir_all(parent)?;
175        }
176
177        let line = serde_json::to_string(event)?;
178        let mut options = OpenOptions::new();
179        options.create(true).append(true);
180
181        #[cfg(unix)]
182        {
183            use std::os::unix::fs::OpenOptionsExt;
184            options.mode(0o600);
185        }
186
187        let mut file = options.open(&self.path)?;
188        writeln!(file, "{line}")?;
189        file.sync_data()?;
190
191        #[cfg(unix)]
192        {
193            use std::os::unix::fs::PermissionsExt;
194            let _ = fs::set_permissions(&self.path, std::fs::Permissions::from_mode(0o600));
195        }
196
197        if self.mode == LegacyStorageMode::Rolling {
198            self.trim_to_last_entries()?;
199        }
200
201        Ok(())
202    }
203
204    fn trim_to_last_entries(&self) -> anyhow::Result<()> {
205        let raw = fs::read_to_string(&self.path).unwrap_or_default();
206        let lines: Vec<&str> = raw
207            .lines()
208            .map(str::trim)
209            .filter(|line| !line.is_empty())
210            .collect();
211
212        if lines.len() <= self.max_entries {
213            return Ok(());
214        }
215
216        let keep_from = lines.len().saturating_sub(self.max_entries);
217        let kept = &lines[keep_from..];
218        let mut rewritten = kept.join("\n");
219        rewritten.push('\n');
220
221        let tmp = self.path.with_extension(format!(
222            "tmp.{}.{}",
223            std::process::id(),
224            Utc::now().timestamp_nanos_opt().unwrap_or_default()
225        ));
226        fs::write(&tmp, rewritten)?;
227
228        #[cfg(unix)]
229        {
230            use std::os::unix::fs::PermissionsExt;
231            let _ = fs::set_permissions(&tmp, std::fs::Permissions::from_mode(0o600));
232        }
233
234        fs::rename(tmp, &self.path)?;
235        Ok(())
236    }
237}

239static TRACE_LOGGER: LazyLock<RwLock<Option<Arc<LegacyTraceLogger>>>> =
240    LazyLock::new(|| RwLock::new(None));
241
242/// Record a runtime trace event in the legacy JSONL shape.
243pub fn record_event(
244    event_type: &str,
245    channel: Option<&str>,
246    provider: Option<&str>,
247    model: Option<&str>,
248    turn_id: Option<&str>,
249    success: Option<bool>,
250    message: Option<&str>,
251    payload: Value,
252) {
253    let logger = TRACE_LOGGER
254        .read()
255        .unwrap_or_else(|e| e.into_inner())
256        .clone();
257    let Some(logger) = logger else {
258        return;
259    };
260
261    let event = LegacyTraceEvent {
262        id: Uuid::new_v4().to_string(),
263        timestamp: Local::now().to_rfc3339(),
264        event_type: event_type.to_string(),
265        channel: channel.map(str::to_string),
266        provider: provider.map(str::to_string),
267        model: model.map(str::to_string),
268        turn_id: turn_id.map(str::to_string),
269        success,
270        message: message.map(str::to_string),
271        payload,
272    };
273
274    if let Err(err) = logger.append(&event) {
275        // The legacy trace writer failed to persist. Report on the zeroclaw_log
276        // pipeline (a separate sink from the failing LegacyTraceLogger; this
277        // function is the only writer to the latter, so no recursion).
278        ::zeroclaw_log::record!(
279            WARN,
280            ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Note)
281                .with_outcome(::zeroclaw_log::EventOutcome::Failure)
282                .with_attrs(::serde_json::json!({ "error": format!("{err}") })),
283            "Failed to write runtime trace event"
284        );
285    }
286}

288/// Record a "turn cancelled" trace event from the gateway webhook path.
289///
290/// `iterations_completed` and `tool_calls_executed` are best-effort counters;
291/// the current `run_tool_call_loop` does not return them, so MVP callers may
292/// pass 0. The fields are reserved in the trace payload for future plumbing.
293pub fn record_turn_cancelled(
294    session_id: &str,
295    iterations_completed: u32,
296    tool_calls_executed: u32,
297    reason: &str,
298) {
299    record_event(
300        "turn_cancelled",
301        None,
302        None,
303        None,
304        None,
305        None,
306        None,
307        serde_json::json!({
308            "session_id": session_id,
309            "iterations_completed": iterations_completed,
310            "tool_calls_executed": tool_calls_executed,
311            "reason": reason,
312        }),
313    );
314}

316#[cfg(test)]
317mod tests {
318    use super::*;
319    use zeroclaw_config::schema::ObservabilityConfig;
320
321    fn test_observability_config(dir: &Path) -> ObservabilityConfig {
322        ObservabilityConfig {
323            log_persistence: "rolling".to_string(),
324            log_persistence_path: dir.join("trace.jsonl").to_string_lossy().into_owned(),
325            log_persistence_max_entries: 2,
326            ..ObservabilityConfig::default()
327        }
328    }
329
330    #[test]
331    fn legacy_record_event_writes_legacy_shape_and_rolls() {
332        let tmp = tempfile::tempdir().unwrap();
333        let cfg = test_observability_config(tmp.path());
334        init_from_config(&cfg, tmp.path());
335
336        for i in 0..4 {
337            record_event(
338                "context_state",
339                Some("webhook"),
340                Some("opencode-go"),
341                Some("deepseek-v4-flash"),
342                None,
343                Some(true),
344                None,
345                serde_json::json!({ "i": i, "tokens_before": 100 + i }),
346            );
347        }
348
349        let raw = std::fs::read_to_string(tmp.path().join("trace.jsonl")).unwrap();
350        let lines: Vec<&str> = raw.lines().filter(|l| !l.trim().is_empty()).collect();
351        assert_eq!(lines.len(), 2, "rolling mode keeps last max_entries lines");
352
353        let last: serde_json::Value = serde_json::from_str(lines[1]).unwrap();
354        assert_eq!(last["event_type"], "context_state");
355        assert_eq!(last["channel"], "webhook");
356        assert_eq!(last["payload"]["tokens_before"], 103);
357        assert!(last["timestamp"].is_string(), "legacy key is `timestamp`");
358        assert!(
359            last.get("@timestamp").is_none(),
360            "must NOT be the new LogEvent shape"
361        );
362    }
